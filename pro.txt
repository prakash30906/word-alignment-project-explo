def tokenize_hindi(sentence: str):
    """
    Tokenizes Hindi properly — splits on spaces but also
    handles postpositions (को, में, ने, का, की, के) that
    are written joined to the word in some encodings.
    Returns a clean word list.
    """
    import re
    # Normalize unicode, split on whitespace
    tokens = sentence.strip().split()
    cleaned = []
    for t in tokens:
        # Remove any trailing punctuation (।,.)
        t = re.sub(r'[।\.\,\?\!]+$', '', t)
        if t:
            cleaned.append(t)
    return cleaned

def tokenize_english(sentence: str):

            cleaned.append(t)
    return cleaned

def get_subword_to_word_map(tokens, word_list):
    """Map each subword token index → original word index."""
    mapping = []
    word_idx = 0
    reconstructed = ""
    for token in tokens:
        clean = token.replace("##", "").replace("▁", "")
        mapping.append(word_idx)
        reconstructed += clean
        if reconstructed == word_list[word_idx].lower():
            word_idx += 1
            reconstructed = ""
    return mapping


def itermax(sim_matrix, max_iter=50):
    """
    Itermax algorithm: iteratively removes the best match,
    enabling many-to-many alignments. Better than simple
    argmax for morphologically rich languages like Hindi.
    """
    alignment = set()
    sim = sim_matrix.copy()
    
    for _ in range(max_iter):
        if sim.max() < 1e-6:
            break
        src_idx, tgt_idx = np.unravel_index(sim.argmax(), sim.shape)
        alignment.add((int(src_idx), int(tgt_idx)))
        sim[src_idx, :] = 0
        sim[:, tgt_idx] = 0
    
    return alignment


def bidirectional_sym(sim_matrix, threshold=0.3):
    """
    Bidirectional softmax symmetrization:
    - Forward: for each source word, softmax over target words
    - Backward: for each target word, softmax over source words
    - Keep alignment if BOTH directions agree above threshold
    This handles one-to-many and many-to-one naturally.
    """
    fwd = F.softmax(torch.tensor(sim_matrix) * 5.0, dim=1).numpy()
    bwd = F.softmax(torch.tensor(sim_matrix) * 5.0, dim=0).numpy()
    
    sym = (fwd > threshold) & (bwd > threshold)
    pairs = set(zip(*np.where(sym)))
    return {(int(i), int(j)) for i, j in pairs}


def get_labse_word_embeddings(words, tokenizer, model, layer=8):
    """
    Get word-level embeddings from LaBSE by averaging subword tokens.
    Uses layer 8 (empirically best for alignment tasks).
    """
    encoding = tokenizer(
        words,
        is_split_into_words=True,
        return_tensors="pt",
        padding=True,
        truncation=True,
        return_offsets_mapping=False
    )
    
    word_ids = encoding.word_ids()  # maps each token → word index
    
    with torch.no_grad():
        outputs = model(**encoding, output_hidden_states=True)
        hidden = outputs.hidden_states[layer][0]  # (seq_len, hidden_dim)
    
    # Average subword embeddings per word
    word_embeddings = []
    for word_idx in range(len(words)):
        token_mask = [i == word_idx for i in word_ids if i is not None]
        # Get token positions for this word
        positions = [i for i, wid in enumerate(word_ids) if wid == word_idx]
        if positions:
            word_emb = hidden[positions].mean(dim=0)
        else:
            word_emb = torch.zeros(hidden.shape[-1])
        word_embeddings.append(word_emb)
    
    return torch.stack(word_embeddings)  # (num_words, hidden_dim)


def align_words(src_sentence: str, tgt_sentence: str, method="itermax"):
    """
    Full alignment pipeline:
    1. Tokenize both sentences properly
    2. Get LaBSE word-level embeddings for each
    3. Compute cosine similarity matrix
    4. Apply bidirectional symmetrization + Itermax
    5. Return sorted (src_idx, tgt_idx) pairs
    """
    src_words = tokenize_english(src_sentence)
    tgt_words = tokenize_hindi(tgt_sentence)
    
    # Get LaBSE embeddings (word-level)
    src_emb = get_labse_word_embeddings(src_words, labse_tokenizer, labse_model)
    tgt_emb = get_labse_word_embeddings(tgt_words, labse_tokenizer, labse_model)
    
    # Cosine similarity matrix (src_len x tgt_len)
    src_norm = F.normalize(src_emb, dim=-1)
    tgt_norm = F.normalize(tgt_emb, dim=-1)
    sim_matrix = torch.matmul(src_norm, tgt_norm.T).numpy()
    
    # Symmetrized alignment
    sym_pairs = bidirectional_sym(sim_matrix, threshold=0.25)
    
    # Itermax on top of sym for many-to-many
    iter_pairs = itermax(sim_matrix)
    
    # Union: take union of both, giving best coverage
    all_pairs = sym_pairs | iter_pairs
    
    return sorted(all_pairs), src_words, tgt_words, sim_matrix

from transformers import AutoTokenizer, AutoModelForSeq2SeqLM

class Translator:
    MODEL_NAME = "facebook/nllb-200-distilled-600M"

    def __init__(self):
        self.tokenizer = AutoTokenizer.from_pretrained(self.MODEL_NAME)
        self.model = AutoModelForSeq2SeqLM.from_pretrained(self.MODEL_NAME)
        self.model.eval()
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)

    def translate(self, sentence: str) -> str:
        if not sentence.strip():
            return ""

        inputs = self.tokenizer(
            sentence,
            return_tensors="pt",
            padding=True,
            truncation=True,
            max_length=256
        ).to(self.device)

        with torch.no_grad():
            output_tokens = self.model.generate(
                **inputs,
                forced_bos_token_id=self.tokenizer.convert_tokens_to_ids("hin_Deva"),
                max_length=512,
                num_beams=5,
                early_stopping=True,
            )

        return self.tokenizer.batch_decode(
            output_tokens,
            skip_special_tokens=True
        )[0]

# alignment/nlp_engine/aligner.py
import torch
import torch.nn.functional as F
import numpy as np
from transformers import AutoTokenizer, AutoModel
from collections import defaultdict
import re

class Aligner:
    MODEL_NAME = "setu4993/LaBSE"

    def __init__(self):
        self.tokenizer = AutoTokenizer.from_pretrained(self.MODEL_NAME)
        self.model = AutoModel.from_pretrained(self.MODEL_NAME)
        self.model.eval()
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)

    # ── tokenizers ──────────────────────────────────────────────
    def _tokenize(self, sentence: str) -> list:
        return [re.sub(r'[।\.\,\?\!]+$', '', w)
                for w in sentence.strip().split() if w]

    # ── embeddings ───────────────────────────────────────────────
    def _word_embeddings(self, words: list, layer: int = 8) -> torch.Tensor:
        enc = self.tokenizer(
            words,
            is_split_into_words=True,
            return_tensors="pt",
            padding=True,
            truncation=True,
        ).to(self.device)

        word_ids = enc.word_ids()

        with torch.no_grad():
            hidden = self.model(**enc, output_hidden_states=True)\
                         .hidden_states[layer][0]

        word_embs = []
        for idx in range(len(words)):
            positions = [i for i, wid in enumerate(word_ids) if wid == idx]
            word_embs.append(hidden[positions].mean(dim=0) if positions
                             else torch.zeros(hidden.shape[-1], device=self.device))
        return torch.stack(word_embs)

    # ── alignment algorithms ──────────────────────────────────────
    @staticmethod
    def _itermax(sim: np.ndarray, max_iter: int = 60) -> set:
        pairs, m = set(), sim.copy()
        for _ in range(max_iter):
            if m.max() < 1e-6:
                break
            i, j = np.unravel_index(m.argmax(), m.shape)
            pairs.add((int(i), int(j)))
            m[i, :] = 0
            m[:, j] = 0
        return pairs

    @staticmethod
    def _bidir_sym(sim: np.ndarray, threshold: float = 0.25) -> set:
        t = torch.tensor(sim) * 5.0
        fwd = F.softmax(t, dim=1).numpy()
        bwd = F.softmax(t, dim=0).numpy()
        mask = (fwd > threshold) & (bwd > threshold)
        return {(int(i), int(j)) for i, j in zip(*np.where(mask))}

    # ── public API ────────────────────────────────────────────────
    def align(self, src_sentence: str, tgt_sentence: str) -> dict:
        src_words = self._tokenize(src_sentence)
        tgt_words = self._tokenize(tgt_sentence)

        src_emb = F.normalize(self._word_embeddings(src_words), dim=-1)
        tgt_emb = F.normalize(self._word_embeddings(tgt_words), dim=-1)
        sim = torch.matmul(src_emb, tgt_emb.T).cpu().numpy()

        pairs = sorted(self._bidir_sym(sim) | self._itermax(sim))

        # Group for display
        src_to_tgt = defaultdict(list)
        tgt_to_src = defaultdict(list)
        for i, j in pairs:
            src_to_tgt[i].append(j)
            tgt_to_src[j].append(i)

        alignment_rows = []
        for i in sorted(src_to_tgt):
            alignment_rows.append({
                "en_idx":    i,
                "en_word":   src_words[i],
                "hi_indices": src_to_tgt[i],
                "hi_words":  [tgt_words[j] for j in sorted(src_to_tgt[i])],
                "many_to_many": len(src_to_tgt[i]) > 1,
            })

        unaligned_hi = [
            {"idx": j, "word": tgt_words[j]}
            for j in range(len(tgt_words))
            if j not in {j2 for _, j2 in pairs}
        ]

        return {
            "src_words":       src_words,
            "tgt_words":       tgt_words,
            "pairs":           pairs,
            "alignment_rows":  alignment_rows,
            "unaligned_hindi": unaligned_hi,
            "sim_matrix":      sim.tolist(),   # for heatmap
        }

class AlignView(APIView):
    """
    POST /api/align/
    Body: { "english": "...", "hindi": "..." (optional) }
    Returns full alignment result + saves to DB.
    """

    def post(self, request):
        serializer = AlignmentRequestSerializer(data=request.data)
        if not serializer.is_valid():
            return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

        english = serializer.validated_data['english'].strip()
        hindi   = serializer.validated_data.get('hindi', '').strip()

        # Get models from app config (loaded once at startup)
        app_config  = apps.get_app_config('alignment')
        translator  = app_config.translator
        aligner     = app_config.aligner

        if not translator or not aligner:
            return Response(
                {"error": "NLP models not loaded. Check server logs."},
                status=status.HTTP_503_SERVICE_UNAVAILABLE
            )

        try:
            t0 = time.time()

            # Step 1: translate if Hindi not provided
            if not hindi:
                hindi = translator.translate(english)

            # Step 2: align
            result = aligner.align(english, hindi)

            elapsed = round(time.time() - t0, 2)

            # Step 3: save to DB
            db_obj = AlignmentResult.objects.create(
                english_sentence  = english,
                hindi_sentence    = hindi,
                alignment_pairs   = result['pairs'],
                similarity_matrix = result['sim_matrix'],
            )

            return Response({
                "id":              db_obj.id,
                "english":         english,
                "hindi":           hindi,
                "src_words":       result['src_words'],
                "tgt_words":       result['tgt_words'],
                "alignment_rows":  result['alignment_rows'],
                "unaligned_hindi": result['unaligned_hindi'],
                "sim_matrix":      result['sim_matrix'],
                "elapsed_sec":     elapsed,
            })

        except Exception as e:
            logger.exception("Alignment failed")
            return Response({"error": str(e)}, status=status.HTTP_500_INTERNAL_SERVER_ERROR)

        return Response(AlignmentResultSerializer(results, many=True).data)

function drawHeatmap(matrix, srcWords, tgtWords) {
  const canvas = document.getElementById('heatmap-canvas');
  const ctx    = canvas.getContext('2d');

  const CELL   = 48;
  const MARGIN = 120;

  canvas.width  = MARGIN + tgtWords.length * CELL;
  canvas.height = MARGIN + srcWords.length * CELL;

  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.font = '11px monospace';

  // Draw cells
  matrix.forEach((row, i) => {
    row.forEach((val, j) => {
      const intensity = Math.max(0, Math.min(1, val));
      // Blue-to-red colorscale
      const r = Math.round(255 * intensity);
      const b = Math.round(255 * (1 - intensity));
      ctx.fillStyle = `rgb(${r}, 80, ${b})`;
      ctx.fillRect(MARGIN + j * CELL, MARGIN + i * CELL, CELL - 2, CELL - 2);

      // Value text
      ctx.fillStyle = intensity > 0.5 ? '#fff' : '#222';
      ctx.fillText(val.toFixed(2),
        MARGIN + j * CELL + 4,
        MARGIN + i * CELL + CELL / 2 + 4);
    });
  });

  // Hindi column labels (top)
  ctx.fillStyle = '#333';
  tgtWords.forEach((w, j) => {
    ctx.save();
    ctx.translate(MARGIN + j * CELL + CELL / 2, MARGIN - 8);
    ctx.rotate(-Math.PI / 4);
    ctx.fillText(w, 0, 0);
    ctx.restore();
  });

  // English row labels (left)
  srcWords.forEach((w, i) => {
    ctx.fillStyle = '#333';
    ctx.fillText(w,
      4,
      MARGIN + i * CELL + CELL / 2 + 4);
  });
}

What ALLOWED_HOSTS actually controls
Django checks the Host header of every incoming request against this list. If the host doesn't match → Django returns a 400 error. That's it. It's a security filter, not a network binding.

How to make it accessible — 3 scenarios
Scenario 1 — Access from other devices on your LAN (same WiFi)
Just add your local IP to .env:

env
ALLOWED_HOSTS=localhost,127.0.0.1,192.168.1.105
Run with:

bash
RUN_MAIN=true python manage.py runserver 0.0.0.0:8000
Now anyone on your WiFi can open http://192.168.1.105:8000 and use it. Find your IP with ip a (Linux) or ipconfig (Windows).

Scenario 2 — Deploy on a VPS / cloud server (DigitalOcean, AWS, etc.)
env
SECRET_KEY=a-strong-random-key
DEBUG=False
ALLOWED_HOSTS=yourdomain.com,www.yourdomain.com,YOUR_SERVER_IP
Then run with gunicorn + nginx:

bash
# gunicorn as app server
GUNICORN=true gunicorn nlp_lab.wsgi:application --workers 1 --timeout 120 --bind 127.0.0.1:8000

# nginx proxies public traffic → gunicorn
# (nginx config shown below)
Nginx config (/etc/nginx/sites-available/nlp_lab):

nginx
server {
    listen 80;
    server_name yourdomain.com;

    location /static/ {
        alias /path/to/your/project/static/;
    }

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 120s;   # important — model inference takes time
    }
}
Scenario 3 — Quick public URL for testing (no server needed)
Use ngrok to tunnel your local server to a public URL instantly:

bash
# Terminal 1 — run Django
RUN_MAIN=true python manage.py runserver 8000

# Terminal 2 — expose it publicly
ngrok http 8000
ngrok gives you a URL like https://abc123.ngrok.io. Add it to .env:

env
ALLOWED_HOSTS=localhost,127.0.0.1,abc123.ngrok.io
Anyone in the world can now access your NLP Lab via that URL.

Summary
Setting	Accessible from
ALLOWED_HOSTS=localhost + runserver	Only your own machine
ALLOWED_HOSTS=192.168.x.x + 0.0.0.0:8000	Everyone on your WiFi
ALLOWED_HOSTS=yourdomain.com + nginx + gunicorn	The entire internet
ngrok tunnel	The entire internet, zero server setup
For an NLP Lab website, the VPS + nginx + gunicorn route (Scenario 2) is the right production path. DigitalOcean's basic $6/month droplet works fine, though you'll want at least 4GB RAM since NLLB + LaBSE together use ~3GB.
